command [--config path/to/config] config locate  # определяет текущее место конфига
                                         init  # генерирует конфиг
                                         validate  # валидирует конфиг
                                         keygen [-t ed25519]  # примечание: ed25519 сейчас по умолчанию, другие вариации будут введены позже
                                         show  # выводит весь конфиг
                                         get  # получает значение из конфига
                                         set [--force]  # выставляет значение в конфиге, проводит валидацию и сохраняет, если прошла (если стоит force сохраняет, даже если валидация не прошла)
                                         del [--force]  # удаляет из конфига, проводит валидацию и сохраняет, если прошла (если стоит force сохраняет, даже если валидация не прошла)
